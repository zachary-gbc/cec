DROP TABLE IF EXISTS cec_Entries;
CREATE TABLE cec_Entries(
  E_ID VARCHAR(100) NOT NULL,
  E_Start DATETIME NOT NULL,
  E_End DATETIME NOT NULL,
  E_Description TEXT NULL,
  E_Summary TEXT NULL,
  E_Location TEXT NULL,
  E_Recurring VARCHAR(50) NULL,
  E_Viewed CHAR(1) NOT NULL DEFAULT 'N',
  E_ScriptNote VARCHAR(50) NULL,
  PRIMARY KEY(E_ID)
);
